874

876  // Search for the first string delimiter
877  while(!inString) {

879  if(cmdData[i + startChar] == STRING_DELIMITER) {
880  i++;
881  inString = true;

883  else {

886  // Exit with an empty string

890  i++;

896  // Collect the string characters
897  while(inString) {
898  // Check if the current char points to the first string delimiter
899  if(cmdData[i + startChar] == STRING_DELIMITER) {
900  inString = false;
901  } // String delimieter found ?

905  // Exit with the collected string
