779  String temp = "";
780  int i = 0;
781  bool inString = false;
782 
783  // ========================================== FIRST STRING DELIMITER
784  // Search for the first string delimiter
785  while(!inString) {
786  // Check if the current char points to the first string delimiter
787  if(cmd.cmdData[i + startChar] == STRING_DELIMITER) {
788  i++;
789  inString = true;
790  } // String delimieter found ?
791  else {
792  // Check for the end of string
793  if(cmd.cmdData[i + startChar] == MAX_CMD_LEN) {
794  // Exit with an empty string
795  return "";
796  } // Reached the end of the command string ?
797  else {
798  i++;
799  } // Search to next character
800  } // Delimiter not found
801  } // While out of string
802 
803  // ========================================== UNTIL SECOND STRING DELIMITER
804  // Collect the string characters
805  while(inString) {
806  // Check if the current char points to the first string delimiter
807  if(cmd.cmdData[i + startChar] == STRING_DELIMITER) {
808  inString = false;
809  } // String delimieter found ?
810  else {
811  // Check for the end of string
812  if(cmd.cmdData[i + startChar] == MAX_CMD_LEN) {
813  // Exit with the collected string
814  return temp;
815  } // Reached the end of the command string ?
816  else {
817  // Collect the character in the string and move to next
818  temp.concat(cmd.cmdData[i + startChar]);
819  i++;
820  } // Search to next character
821  } // Delimiter not found
822  } // While inside the string
823 
824  // Found both delimiters and collected the string.
